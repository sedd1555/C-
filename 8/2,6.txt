#include <iostream>
using namespace std;
int main(void) {
    long c0;
    int elemetn;
    
   while (true){
    cin >> c0;
    while (c0!=1){
      if (c0%2==0){
        c0=c0/2;
      }else if (c0%2!=0){
        c0=3*c0+1;
      }
      cout << c0 << endl;
      elemetn++;
    }
    cout << "Steps = ";
    cout << elemetn << endl;
    elemetn = 0;
  }
    
    
    return 0;
}
